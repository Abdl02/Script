python
# auth.py

import requests

class AuthManager:
    def __init__(self, auth_url, client_id, client_secret):
        self.auth_url = auth_url
        self.client_id = client_id
        self.client_secret = client_secret
        self.token = None

    def get_token(self):
        """Request a token from the auth server."""
        payload = {
            "client_id": self.client_id,
            "client_secret": self.client_secret,
            "grant_type": "client_credentials"  # or whatever your API expects
        }
        
        try:
            response = requests.post(self.auth_url, data=payload)
            response.raise_for_status()  # Raises exception if the call failed
            self.token = response.json().get("access_token")
            if not self.token:
                raise Exception("Token not found in response!")
            return self.token
        except Exception as e:
            print(f"[Auth Error] {e}")
            return None

    def get_auth_header(self):
        """Return Authorization header ready to use."""
        if not self.token:
            self.get_token()
        return {
            "Authorization": f"Bearer {self.token}"
        }

3. flow/flow_manager.py (Core logic) ✅

python
import yaml
from utils.mapping import Mapper
from utils.dependency_resolver import DependencyResolver
from utils.config_loader import ConfigLoader
import requests

class FlowManager:
    def __init__(self, flow_file):
        self.flow_file = flow_file
        self.load_flow()

    def load_flow(self):
        with open(self.flow_file, "r") as f:
            self.flow_data = yaml.safe_load(f)
        self.environment = self.flow_data.get("environment", {})
        self.requests = self.flow_data.get("requests", [])
        self.saved_data = {}

    def execute_flow(self):
        print(f"Executing flow: {self.flow_data['name']}")
        for req in self.requests:
            self.execute_request(req)

    def execute_request(self, request_info):
        url = self.render(request_info['url'])
        method = request_info['method']
        headers = request_info.get('headers', {})
        body = self.render(request_info.get('body', None))

        if body:
            body = yaml.safe_load(body)  # Load JSON from string if needed

        response = requests.request(method, url, headers=headers, json=body)
        print(f"{method} {url} -> {response.status_code}")

        if 'save_as' in request_info:
            self.saved_data[request_info['save_as']] = response.json()

    def render(self, template):
        if not template:
            return template
        for key, value in {**self.environment, **self.flatten(self.saved_data)}.items():
            template = template.replace(f"{{{{{key}}}}}", str(value))
        return template

    def flatten(self, d, parent_key='', sep='.'):
        items = {}
        for k, v in d.items():
            new_key = parent_key + sep + k if parent_key else k
            if isinstance(v, dict):
                items.update(self.flatten(v, new_key, sep=sep))
            else:
                items[new_key] = v
        return items
